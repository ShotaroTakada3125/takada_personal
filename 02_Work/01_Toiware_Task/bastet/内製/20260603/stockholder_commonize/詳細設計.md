## State 構造

### 【現在の構造】単一キャンペーン向け

```javascript

  entryPasswordLabel: "",
```

**問題**:
- `requestCheckAccount` が単一フラグ → 複数タブで上書きされる

### 【API呼び出しパターン】

全API呼び出し（4つ）で以下の流れを共通化:

```
REQUEST → API通信 → SUCCESS or FAILURE

REQUEST 時点で campaignKey を保持
SUCCESS/FAILURE でも campaignKey を payload に含める
Reducer で campaignKey をキーにして state 更新
```

### ① 申込可否チェック API

```javascript
/**
 * 申込可否チェック
 * API①: GET /api/stockholder_bonus_entry/{keyword}/{term}/{accountId}/account_check
 *
 * @param {object} params { keyword, term, accountId }
 * @param {object} history
 */
export const checkAccount = (params, history) => async (dispatch, getState) => {
  const { keyword, term } = params;
  const campaignKey = generateCampaignKey(keyword, term);

  // accountId をセッションから取得
  const { account } = getState();
  const accountId = account?.accountId;

  if (!accountId) {
    alert('ログインしてください');
    history.replace('/signin');
    return;
  }

  // 不正遷移チェック
  dispatch(checkUnauthorizedNavigation(keyword, term, history));

  // REQUEST アクションをディスパッチ
  dispatch(requestCheckAccount(campaignKey));

  try {
    const result = await request({
      path: `/api/stockholder_bonus_entry/account_check`,
      method: 'GET',
      params: { keyword, term },
    });

    if (result === null || result.status !== 200) {
      if (result && result.res && result.res.direct) {
        alert(result.res.message);
        history.replace("/signin");
      }
      return dispatch(failureCheckAccount(campaignKey, result));
    }

    return dispatch(successCheckAccount(campaignKey, result));
  } catch (error) {
    return dispatch(failureCheckAccount(campaignKey, null));
  }
};

// REQUEST アクション
export const requestCheckAccount = (campaignKey) => ({
  type: c.REQUEST_CHECK_ACCOUNT,
  payload: { campaignKey },
});

// FAILURE アクション
export const failureCheckAccount = (campaignKey, result) => ({
  type: c.FAILURE_CHECK_ACCOUNT,
  payload: {
    campaignKey,
    res: result === null ? null : result.res,
    status: result === null ? null : result.status,
  },
});

// SUCCESS アクション
export const successCheckAccount = (campaignKey, result) => ({
  type: c.SUCCESS_CHECK_ACCOUNT,
  payload: {
    campaignKey,
    ...result.res,  // { campaignName, entryCodeLabel, ... }
  },
});
```

### ③ 入力値検証 API

```javascript
/**
 * 入力値検証
 * API②: POST /api/stockholder_bonus_entry/{keyword}/{term}/{accountId}/entry_check
 *
 * @param {object} params { keyword, term, formData }
 * @param {object} history
 */
export const setBasic = (params, history) => async (dispatch) => {
  const { keyword, term, formData } = params;
  const campaignKey = generateCampaignKey(keyword, term);

  // クライアント側のバリデーション実行（ANAPと同じパターン）
  if (!dispatch(validateStockholderCampaign(keyword, formData))) {
    return;  // バリデーション失敗時は処理中止
  }

  dispatch(putSession(formData));
  dispatch(requestSetBasic(campaignKey));

  try {
    const result = await request({
      path: '/api/stockholder_bonus_entry/entry_check',
      method: 'POST',
      params: { keyword, term },
      data: formData,
    });

    if (result === null || result.status !== 200) {
      return dispatch(failureSetBasic(campaignKey, result));
    }

    history.push(`/${keyword}/${term}/confirm`);
    return dispatch(successSetBasic(campaignKey, result));
  } catch (error) {
    return dispatch(failureSetBasic(campaignKey, null));
  }
};

export const requestSetBasic = (campaignKey) => ({
  type: c.REQUEST_SET_BASIC,
  payload: { campaignKey },
});

export const failureSetBasic = (campaignKey, result) => ({
  type: c.FAILURE_SET_BASIC,
  payload: {
    campaignKey,

    status: result === null ? null : result.status,
  },
});

export const successSetBasic = (campaignKey, result) => ({
  type: c.SUCCESS_SET_BASIC,
  payload: {
    campaignKey,
    mainEmail: result.res.mainEmail,
    fullName: result.res.fullName,

});
```

### ④ 申込 API

```javascript
/**
 * 申込実行
 * API③: POST /api/stockholder_bonus_entry/{keyword}/{term}/{accountId}
 *
 * @param {object} params { keyword, term, formData }
 * @param {function} redirect
 */
export const post = (params, redirect) => async (dispatch, getState) => {
  const { keyword, term, formData } = params;
  const campaignKey = generateCampaignKey(keyword, term);

  dispatch(requestPost(campaignKey));

  try {
    const result = await request({
      path: "/api/stockholder_bonus_entry",
      method: "POST",
      params: { keyword, term },
      data: formData,
    });

    if (result === null || result.status !== 200) {
      return dispatch(failurePost(campaignKey, result));
    }

    sessionStorage.removeItem("rd");

    // 【重要】campaignKey から campaignName を取得
    const { stockholderCampaign } = getState();
    const campaignInfo = stockholderCampaign.campaignMeta[campaignKey];
    const campaignName = campaignInfo?.campaignName || "";

    alert(`${campaignName}のお申込みを確定しました。`);

    redirect();
    return dispatch(successPost(campaignKey));
  } catch (error) {
    return dispatch(failurePost(campaignKey, null));
  }
};

  type: c.REQUEST_POST,

});

});
```

### 【ハンドラー実装】

```javascript
export default function stockholderCampaign(state = initialState, action) {
  switch (action.type) {
    // ==================== 申込可否チェック ====================

    case c.REQUEST_CHECK_ACCOUNT:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: true,
        },
      };

    case c.FAILURE_CHECK_ACCOUNT:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: false,
        },
        errorsByKey: {
          ...state.errorsByKey,
          [action.payload.campaignKey]: {
            ...(state.errorsByKey[action.payload.campaignKey] || {}),
            serverError: action.payload.res === null
              ? {}
              : { ...action.payload.res, httpStatus: action.payload.status },
          },
        },
      };

    case c.SUCCESS_CHECK_ACCOUNT:
      const { campaignKey: ckCheckAccount, ...metaData } = action.payload;
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [ckCheckAccount]: false,
        },
        campaignMeta: {
          ...state.campaignMeta,
          [ckCheckAccount]: {
            campaignName: metaData.campaignName,
            entryCodeLabel: metaData.entryCodeLabel,
            entryPasswordLabel: metaData.entryPasswordLabel,
            accountType: metaData.accountType,
          },
        },
        errorsByKey: {
          ...state.errorsByKey,
          [ckCheckAccount]: {
            clientError: {},
            serverError: {},
          },
        },
      };

    // ==================== フォーム入力 ====================

    case c.INPUT_FORM:
      return {
        ...state,
        formDataByKey: {
          ...state.formDataByKey,
          [action.payload.campaignKey]: {
            ...(state.formDataByKey[action.payload.campaignKey] || {}),
            [action.payload.key]: action.payload.value,
          },
        },
        errorsByKey: {
          ...state.errorsByKey,
          [action.payload.campaignKey]: {
            ...state.errorsByKey[action.payload.campaignKey],
            clientError: deleteProperty(
              action.payload.key,
              state.errorsByKey[action.payload.campaignKey]?.clientError || {}
            ),
          },
        },
      };

    // ==================== バリデーション ====================

    case cc.VALIDATE_BASIC:
      // 【重要】 VALIDATE_BASIC は campaignKey がないため、別途処理が必要
      // または、validateアクションに campaignKey を含めるようにする
      if (action.payload.result) {
        return {
          ...state,
          errorsByKey: {
            ...state.errorsByKey,
            [action.payload.campaignKey]: {
              ...state.errorsByKey[action.payload.campaignKey],
              clientError: deleteProperty(
                action.payload.key,
                state.errorsByKey[action.payload.campaignKey]?.clientError || {}
              ),
            },
          },
        };
      }
      return {
        ...state,
        errorsByKey: {
          ...state.errorsByKey,
          [action.payload.campaignKey]: {
            ...state.errorsByKey[action.payload.campaignKey],
            clientError: {
              ...(state.errorsByKey[action.payload.campaignKey]?.clientError || {}),
              [action.payload.key]: action.payload.message,
            },
          },
        },
      };

    // ==================== 入力値検証 API ====================

    case c.REQUEST_SET_BASIC:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: true,
        },
      };

    case c.FAILURE_SET_BASIC:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: false,
        },
        errorsByKey: {
          ...state.errorsByKey,
          [action.payload.campaignKey]: {
            ...(state.errorsByKey[action.payload.campaignKey] || {}),
            serverError: action.payload.res === null
              ? {}
              : { ...action.payload.res, httpStatus: action.payload.status },
          },
        },
      };

    case c.SUCCESS_SET_BASIC:
      const { campaignKey: ckSetBasic, mainEmail, fullName } = action.payload;
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [ckSetBasic]: false,
        },
        mainEmail,
        fullName,
        errorsByKey: {
          ...state.errorsByKey,
          [ckSetBasic]: {
            clientError: {},
            serverError: {},
          },
        },
      };

    // ==================== 申込 API ====================

    case c.REQUEST_POST:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: true,
        },
      };

    case c.FAILURE_POST:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: false,
        },
        errorsByKey: {
          ...state.errorsByKey,
          [action.payload.campaignKey]: {
            ...(state.errorsByKey[action.payload.campaignKey] || {}),
            serverError: action.payload.res === null ? {} : action.payload.res,
          },
        },
      };

    case c.SUCCESS_POST:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: false,
        },
      };

    // ==================== 申込済みデータ取得 API ====================

    case c.REQUEST_GET:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: true,
        },
      };

    case c.FAILURE_GET:
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [action.payload.campaignKey]: false,
        },
        errorsByKey: {
          ...state.errorsByKey,
          [action.payload.campaignKey]: {
            ...(state.errorsByKey[action.payload.campaignKey] || {}),
            serverError: action.payload.res === null
              ? {}
              : { ...action.payload.res, httpStatus: action.payload.status },
          },
        },
      };

    case c.SUCCESS_GET:
      const { campaignKey: ckGet, ...getData } = action.payload;
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [ckGet]: false,
        },
        formDataByKey: {
          ...state.formDataByKey,
          [ckGet]: {
            entryCode: getData.entryCode || "",
            entryPassword: getData.entryPassword || "",
          },
        },
        registeredMsg: getData.registeredMsg,
        fullName: getData.fullName,
        updateDatetime: getData.updateDatetime,
      };

    // ==================== リセット ====================

    case c.RESET_FLAGS:
      const { campaignKey: ckReset } = action.payload;
      return {
        ...state,
        requestState: {
          ...state.requestState,
          [ckReset]: null,
        },
        errorsByKey: {
          ...state.errorsByKey,
          [ckReset]: {
            clientError: {},
            serverError: {},
          },
        },
      };

    case c.RESET_ALL_FLAGS:
      return initialState;

    default:
      return state;
  }
}
```

### 【Component での使用例】

```javascript
import React from 'react';
import { connect } from 'react-redux';
import * as actions from '../actions/stockholderCampaign';
import * as selectors from '../selectors/stockholderCampaign';

class EntryFormComponent extends React.Component {
  componentDidMount() {
    const { keyword, term } = this.props;
    this.props.checkAccount({ keyword, term });
    this.props.getSession();
  }

  componentWillUnmount() {
    const { keyword, term } = this.props;
    this.props.reset(`${keyword}/${term}`);
  }

  handleInputChange = (e) => {
    const { name, value } = e.target;
    const { keyword, term } = this.props;
    const campaignKey = `${keyword}/${term}`;
    this.props.inputForm(campaignKey, name, value);
  };

  handleSubmit = () => {
    const { keyword, term, formData } = this.props;
    this.props.setBasic({
      keyword,
      term,
      formData,
    }, this.props.history);
  };

  render() {
    const { campaignMeta, formData, errors, isLoading } = this.props;

    return (
      <div>
        <h1>{campaignMeta?.campaignName}</h1>

        {isLoading && <p>読み込み中...</p>}
        {errors?.serverError?.message && <p className="error">{errors.serverError.message}</p>}

        <input
          type="text"
          name="entryCode"
          value={formData.entryCode || ""}
          onChange={this.handleInputChange}
          placeholder={campaignMeta?.entryCodeLabel}
        />
        {errors?.clientError?.entryCode && <p className="error">{errors.clientError.entryCode}</p>}

        <input
          type="password"
          name="entryPassword"
          value={formData.entryPassword || ""}
          onChange={this.handleInputChange}
          placeholder={campaignMeta?.entryPasswordLabel}
        />
        {errors?.clientError?.entryPassword && <p className="error">{errors.clientError.entryPassword}</p>}

        <button onClick={this.handleSubmit} disabled={isLoading}>
          確認
        </button>
      </div>
    );
  }
}

function mapStateToProps({ stockholderCampaign }, { keyword, term }) {
  const campaignKey = `${keyword}/${term}`;

  return {
    campaignMeta: selectors.selectCampaignMeta(stockholderCampaign, campaignKey),
    formData: selectors.selectFormData(stockholderCampaign, campaignKey),
    errors: selectors.selectErrors(stockholderCampaign, campaignKey),
    isLoading: selectors.selectRequestState(stockholderCampaign, campaignKey),
  };
}

function mapDispatchToProps(dispatch) {
  return {
    checkAccount(params, history) {
      dispatch(actions.checkAccount(params, history));
    },
    getSession() {
      dispatch(actions.getSession());
    },
    inputForm(campaignKey, key, value) {
      dispatch(actions.inputForm(campaignKey, key, value));
    },
    setBasic(params, history) {
      dispatch(actions.setBasic(params, history));
    },
    reset(campaignKey) {
      dispatch(actions.reset(campaignKey));
    },
  };
}

export default connect(mapStateToProps, mapDispatchToProps)(EntryFormComponent);
```

## 実装上の注意点

### ⚠️ **バリデーション対応（ANAPパターンを採用）**

**方針**: ANAPと同じく、`setBasic()` 内でキャンペーン固有のバリデーター関数を呼び出す

**1️⃣ キャンペーン固有のバリデーター関数を作成**

```javascript
// client/actions/stockholderCampaign.js

/**
 * StockholderCampaign キャンペーン固有バリデーション
 * 各キャンペーンで validateSetBasicAnap 相当の関数を作成
 *
 * @example
 * // ANAP の場合
 * export const validateSetBasicAnap = (params) => (dispatch) => { ... }
 *
 * // Convano の場合
 * export const validateSetBasicConvano = (params) => (dispatch) => { ... }
 *
 * // StockholderCampaign の場合
 * export const validateStockholderCampaign = (keyword, params) => (dispatch) => {
 *   // keyword に応じた異なるバリデーションを実行
 * }
 */
export const validateStockholderCampaign = (keyword, params) => (dispatch) => {
  // keyword に応じたバリデーター関数を動的に呼び出す

  if (keyword === 'anap') {
    // ANAP固有バリデーション
    const entryCodeAction = v.validateAnapEntryCode("entryCode", params.entryCode);
    dispatch(entryCodeAction);

    const entryPasswordAction = v.validateAnapEntryPassword("entryPassword", params.entryPassword);
    dispatch(entryPasswordAction);

    // ... その他ANAP固有フィールドのバリデーション

    return entryCodeAction.payload.result && entryPasswordAction.payload.result;
  }

  if (keyword === 'convano') {
    // Convano固有バリデーション
    const stockholderNumberAction = v.validateConvanoStockholderNumber("stockholderNumber", params.stockholderNumber);
    dispatch(stockholderNumberAction);

    const entryPasswordAction = v.validateConvanoEntryPassword("entryPassword", params.entryPassword);
    dispatch(entryPasswordAction);

    return stockholderNumberAction.payload.result && entryPasswordAction.payload.result;
  }

  return true;
};
```

**2️⃣ setBasic() でバリデーター関数を呼び出し**

```javascript
export const setBasic = (params, history) => async (dispatch) => {
  const { keyword, term, formData } = params;
  const campaignKey = generateCampaignKey(keyword, term);

  // バリデーション実行（ANAPパターンと同じ）
  if (!dispatch(validateStockholderCampaign(keyword, formData))) {
    return;  // バリデーション失敗時は処理中止
  }

  dispatch(putSession(formData));
  dispatch(requestSetBasic(campaignKey));

  try {
    const result = await request({
      path: '/api/stockholder_bonus_entry/entry_check',
      method: 'POST',
      params: { keyword, term },
      data: formData,
    });

    if (result === null || result.status !== 200) {
      return dispatch(failureSetBasic(campaignKey, result));
    }

    history.push(`/${keyword}/${term}/confirm`);
    return dispatch(successSetBasic(campaignKey, result));
  } catch (error) {
    return dispatch(failureSetBasic(campaignKey, null));
  }
};
```

**メリット**:
- ✅ ANAPパターンとの一貫性を保つ
- ✅ 既存 validate.js に手を入れない
- ✅ キャンペーン固有バリデーションを集約できる
- ✅ keyword から動的にバリデーション処理を分岐可能

### ⚠️ **セッション情報の共有**

セッション取得時は、**共通フィールド**のみ更新（各キャンペーンのformDataには含めない）

```javascript
case SUCCESS_GET_SESSION:
  if (action.payload === undefined) {
    return { ...state, requestGetSession: false };
  }

  // 【重要】セッション情報は共通フィールドのみ
  // 各キャンペーンの formDataByKey には影響しない
  return {
    ...state,
    requestGetSession: false,
    mainEmail: action.payload.mainEmail,
    fullName: action.payload.fullName,
    // ... その他共通フィールド
  };
```

**理由**:  - 各キャンペーンのformData（entryCode, entryPassword など）はキャンペーン個別フィールド
- セッション情報（mainEmail, fullName）は共通フィールド
- 混在させると複数タブ時に問題発生

### ⚠️ **Cleanup 時のフラグリセット（重要）**

**パターン**: ANAPと同じく、クラスコンポーネントの `componentWillUnmount` を使用

```javascript
// Component内（クラスコンポーネント）
class EntryComponent extends React.Component {
  componentDidMount() {
    this.props.checkAccount({ keyword, term });
    this.props.getSession();
  }

  componentWillUnmount() {
    // キャンペーン固有のフラグをリセット
    // 全フラグをリセットするのではなく、該当キャンペーンのみ
    this.props.reset(`${keyword}/${term}`);
  }

  render() {
    // ...
  }
}
```

**Container（index.jsx）:**
```javascript
export function mapDispatchToProps(dispatch) {
  return {
    checkAccount(params, history) {
      dispatch(checkAccount(params, history));
    },
    getSession() {
      dispatch(getSession());
    },
    reset(campaignKey) {
      dispatch(reset(campaignKey));
    },
  };
}
```

**注意**:
- すべてをリセットするのではなく、該当キャンペーンキーのみリセット（他のTabで使用中の可能性）
- `resetAll()` は全キャンペーンリセット時のみ使用
