+import * as c from "../constants/stockholderCampaign";

+/**
+ * 株主優待キャンペーン用 ログイン処理（共通）
+ *
+ * account.js の simpleLogin() を呼び出した後、申込可否チェックを行う
+ * 1. Simplex ログイン（account.js で実行）
+ * 2. 申込可否チェック API呼び出し
+ * 3. チェック結果による分岐
+ *    - NG: エラーアラート表示 → 通常ログイン画面へ遷移
+ *    - OK: 2要素認証画面へ遷移
+ *
+ * @params {object} params ログイン情報 { email, password, redirectTo, cancelTo, isLogined }
+ * @params {object} campaignParams キャンペーン情報 { keyword, term }
+ * @params {object} history React Router の history オブジェクト
+ */
+export const simpleLoginForStockholderCampaign = (params, campaignParams, history) => async (dispatch) => {
+  const { keyword, term } = campaignParams;
+
+  // account.js の simpleLogin() を呼び出し
+  await dispatch(simpleLogin(
+    params,
+    history,
+    async () => {
+      // 申込可否チェック
+      const checkResult = await dispatch(checkAccount({ keyword, term }, history));
+
+      // チェック結果による分岐
+      if (checkResult.payload.status !== 200) {
+        // エラーメッセージを表示して通常ログイン画面へリダイレクト
+        const message = (checkResult.payload.res && checkResult.payload.res.message) || "申し込みの確認に失敗しました。";
+        alert(message);
+        history.replace("/signin");
+        return;
+      }
+
+      // 2要素認証画面へ遷移
+      history.push("/second_auth");
+    }
+  ));
+};
+
+/**
+ * 不正遷移チェック
+ * 直リンクアクセスの場合は口座状況が参照できないので、不正遷移として申込ログイン画面へ
+ *
+ * @params {string} keyword キャンペーン識別子
+ * @params {string} term 期間識別子
+ * @params {object} history リダイレクトに利用
+ */
+export const checkUnauthorizedNavigation = (keyword, term, history) => (dispatch, getState) => {
+  const { account } = getState();
+
+  // 直リンクの場合は口座状況が参照できないので、不正遷移として申込ログイン画面へ
+  if (!account || !account.status) {
+    history.replace(`/${keyword}/${term}/input`);
+  }
+};
+
+/**
+ * 申込可否チェック
+ *
+ * @params {object} params { keyword, term }
+ * @params {object} history リダイレクトに利用
+ */
+export const checkAccount = (params, history) => async (dispatch) => {
+  const { keyword, term } = params;
+  // 不正遷移チェック
+  dispatch(checkUnauthorizedNavigation(keyword, term, history));
+
+  // すでに申込み済みであれば通常ログイン画面へリダイレクト
+  dispatch(requestCheckAccount());
+  const result = await request({
+    path: '/api/stockholder_bonus_entry/account_check',
+    method: 'GET',
+    params,
+  });
+
+  if (result === null || result.status !== 200) {
+    if (result && result.res && result.res.direct) {
+      alert(result.res.message);
+      history.replace("/signin");
+    }
+    return dispatch(failureCheckAccount(result));
+  }
+
+  return dispatch(successCheckAccount(result));
+};
+

+  return {
+    type: c.REQUEST_CHECK_ACCOUNT,
+  };

+export const setBasic = (params, history) => async (dispatch) => {

+  if (!dispatch(validateSetBasic({ keyword, formData }))) {

+    method: 'GET',
+    params,
+  });
+
+  if (result === null || result.status !== 200) {
+    return dispatch(failureSetBasic(result));
+  }

+  history.push("/" + keyword + "/" + term + "/confirm");

+ * 新規キャンペーン追加時は constants/stockholderCampaign.js に追記するのみ
+ *
+ * @params {object} params { keyword, formData }
+ */
+export const validateSetBasic = (params) => (dispatch) => {
+  const { keyword, formData } = params;
+  // validation 定義を取得
+  const schema = c.CAMPAIGN_VALIDATION_SCHEMAS[keyword];
+  if (!schema) {
+    throw new Error(`Invalid campaign keyword: ${keyword}`);
+  }
+
+  // バリデーション実行
+  let hasError = false;
+  for (const field of schema.fields) {
+    const { key, validator: validatorName } = field;
+
+    // validator 関数を取得
+    const validatorFn = v[validatorName];
+    if (!validatorFn) {
+      throw new Error(
+        `Validator function not found: ${validatorName} (campaign: ${keyword})`
+      );
+    }
+
+    // validation 関数を実行
+    const action = validatorFn(key, formData[key]);
+    dispatch(action);
+
+    if (!action.payload.result) {
+      hasError = true;
+    }
+  }
+
+  return !hasError;
+};
